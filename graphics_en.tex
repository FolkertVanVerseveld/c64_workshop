\begin{frame}[fragile]{Simple Graphics Program - BASIC}

\begin{lstlisting}
10 VIC=53248
20 C=PEEK(VIC+32)AND15
30 PRINT"KLEURCODE=",C
40 C=(C+3)AND15
50 POKEVIC+32,C
60 PRINT"KLEURCODE=",C
\end{lstlisting}

\end{frame}
